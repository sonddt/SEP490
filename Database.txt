-- Workbench: bôi đen một đoạn thì DATABASE() có thể lệch — luôn USE shuttle_up; hoặc chọn schema ở panel trái.

CREATE TABLE users (
    id CHAR(36) PRIMARY KEY,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255) NOT NULL,
    phone_number VARCHAR(20),
    about TEXT,
    address VARCHAR(500),
    district VARCHAR(100),
    province VARCHAR(100),
    gender VARCHAR(20),
    date_of_birth DATE,
    skill_level VARCHAR(50) NULL,
    play_purpose VARCHAR(100) NULL,
    play_frequency VARCHAR(50) NULL,
    is_personalized BOOLEAN DEFAULT FALSE,
    is_active BOOLEAN DEFAULT TRUE,
    blocked_at DATETIME,
    blocked_reason TEXT,
    blocked_by CHAR(36),
    avatar_file_id CHAR(36),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE roles (
    id CHAR(36) PRIMARY KEY,
    name VARCHAR(50) UNIQUE NOT NULL
);

CREATE TABLE user_privacy_settings (
    user_id CHAR(36) PRIMARY KEY,
    allow_find_by_email TINYINT(1) NOT NULL DEFAULT 1,
    allow_find_by_phone TINYINT(1) NOT NULL DEFAULT 1,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE

CREATE TABLE user_blocks (
    blocker_id CHAR(36) NOT NULL,
    blocked_id CHAR(36) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (blocker_id, blocked_id),
    FOREIGN KEY (blocker_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (blocked_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_user_blocks_blocked (blocked_id)
);

CREATE TABLE friendships (
    id CHAR(36) PRIMARY KEY,
    user_low_id CHAR(36) NOT NULL,
    user_high_id CHAR(36) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_friend_pair (user_low_id, user_high_id),
    FOREIGN KEY (user_low_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (user_high_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE friend_requests (
    id CHAR(36) PRIMARY KEY,
    from_user_id CHAR(36) NOT NULL,
    to_user_id CHAR(36) NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'PENDING',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    responded_at DATETIME NULL,
    INDEX idx_friend_req_to_status (to_user_id, status),
    INDEX idx_friend_req_from_status (from_user_id, status),
    FOREIGN KEY (from_user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (to_user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE venues (
    id CHAR(36) PRIMARY KEY,
    owner_user_id CHAR(36),
    name VARCHAR(255) NOT NULL,
    address TEXT NOT NULL,
    lat DECIMAL(10,8),
    lng DECIMAL(11,8),
    description TEXT,
    weekly_discount_percent DECIMAL(5,2) DEFAULT 0,
    monthly_discount_percent DECIMAL(5,2) DEFAULT 0,
    payment_bank_name VARCHAR(100) NULL COMMENT 'Display name e.g. Vietcombank',
    payment_bank_bin VARCHAR(20) NULL COMMENT '6-digit VietQR bank ID',
    payment_account_number VARCHAR(50) NULL,
    payment_account_holder VARCHAR(255) NULL,
    payment_transfer_note_template VARCHAR(500) NULL DEFAULT '[SĐT] - [Tên sân] - [Ngày]',
    cancel_allowed TINYINT(1) NOT NULL DEFAULT 1,
    cancel_before_minutes INT NOT NULL DEFAULT 120,
    refund_type VARCHAR(20) NOT NULL DEFAULT 'NONE' COMMENT 'NONE|PERCENT|FULL',
    refund_percent DECIMAL(5,2) NULL,
    thumbnail_file_id CHAR(36),
    contact_name VARCHAR(255),
    contact_phone VARCHAR(50),
    includes JSON NULL COMMENT 'Mảng chuỗi — những gì khách được sử dụng khi thuê sân',
    rules JSON NULL COMMENT 'Mảng chuỗi — các quy định tại cơ sở',
    amenities JSON NULL COMMENT 'Mảng key chuỗi — tiện ích có tại cơ sở (parking, wifi, ...)',
    is_active BOOLEAN DEFAULT TRUE,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (owner_user_id) REFERENCES users(id),
    FOREIGN KEY (thumbnail_file_id) REFERENCES files(id)
);

CREATE TABLE courts (
    id CHAR(36) PRIMARY KEY,
    venue_id CHAR(36),
    name VARCHAR(100),
    group_name VARCHAR(100) NULL,
    surface VARCHAR(50),
    max_guest INT,
    description TEXT,
    status VARCHAR(50) DEFAULT 'ACTIVE',
    is_active BOOLEAN DEFAULT TRUE,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (venue_id) REFERENCES venues(id) ON DELETE CASCADE
);

CREATE TABLE venue_open_hours (
    id CHAR(36) PRIMARY KEY,
    venue_id CHAR(36),
    day_of_week INT,
    open_time TIME,
    close_time TIME,
    FOREIGN KEY (venue_id) REFERENCES venues(id) ON DELETE CASCADE
);

-- Bài đăng trang Nổi bật (Admin + Manager quảng cáo)
CREATE TABLE featured_posts (
    id CHAR(36) PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    excerpt VARCHAR(500) NULL,
    body TEXT NULL,
    cover_image_url VARCHAR(2048) NULL,
    link_url VARCHAR(2048) NULL,
    is_published TINYINT(1) NOT NULL DEFAULT 0,
    display_from DATETIME NULL COMMENT 'NULL = không giới hạn đầu',
    display_until DATETIME NULL COMMENT 'NULL = không giới hạn cuối',
    sort_order INT NOT NULL DEFAULT 0,
    author_user_id CHAR(36) NOT NULL,
    author_role VARCHAR(20) NOT NULL COMMENT 'ADMIN | MANAGER',
    venue_id CHAR(36) NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (author_user_id) REFERENCES users(id),
    FOREIGN KEY (venue_id) REFERENCES venues(id) ON DELETE SET NULL,
    INDEX idx_featured_posts_published_dates (is_published, display_from, display_until),
    INDEX idx_featured_posts_author (author_user_id)
);

CREATE TABLE court_prices (

    end_time TIME,
    price DECIMAL(15,2),
    is_weekend BOOLEAN DEFAULT FALSE,

CREATE TABLE court_open_hours (
    id CHAR(36) PRIMARY KEY,
    court_id CHAR(36),
    day_of_week INT,
    open_time TIME NULL,
    close_time TIME NULL,
    FOREIGN KEY (court_id) REFERENCES courts(id) ON DELETE CASCADE,
    UNIQUE KEY uq_court_open_hours (court_id, day_of_week)
);

CREATE TABLE venue_coupons (
    id CHAR(36) PRIMARY KEY,
    venue_id CHAR(36) NOT NULL,
    code VARCHAR(50) NOT NULL,
    discount_type VARCHAR(20) DEFAULT 'PERCENT' COMMENT 'PERCENT|FIXED_AMOUNT',
    discount_value DECIMAL(15,2) NOT NULL,
    min_booking_value DECIMAL(15,2) DEFAULT 0,
    max_discount_amount DECIMAL(15,2) NULL,
    start_date DATETIME NOT NULL,
    end_date DATETIME NOT NULL,

    used_count INT DEFAULT 0,

    FOREIGN KEY (venue_id) REFERENCES venues(id) ON DELETE CASCADE,
    UNIQUE KEY uq_venue_coupon (venue_id, code)
);

-- ======================================================================
-- BOOKING
-- ======================================================================
-- booking_series.recurrence_rule_json (ứng dụng quy ước, không enforce ở DB):
--   WEEKLY: { "type":"WEEKLY", "daysOfWeek":[...], "sessionStart","sessionEnd", "courtId" }
--   FLEXIBLE: { "type":"FLEXIBLE", "itemCount": N } — range_start_date/range_end_date = min/max ngày từ booking_items

CREATE TABLE booking_series (
    id CHAR(36) PRIMARY KEY,
    user_id CHAR(36) NOT NULL,
    venue_id CHAR(36) NOT NULL,
    recurrence_rule_json TEXT NOT NULL,
    range_start_date DATE NOT NULL,
    range_end_date DATE NOT NULL,
    status VARCHAR(50) DEFAULT 'PENDING',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (venue_id) REFERENCES venues(id),
    INDEX booking_series_user_id (user_id),
    INDEX booking_series_venue_id (venue_id)
);

CREATE TABLE bookings (
    id CHAR(36) PRIMARY KEY,
    user_id CHAR(36),
    venue_id CHAR(36),
    series_id CHAR(36) NULL,
    coupon_id CHAR(36) NULL,
    status VARCHAR(50) DEFAULT 'PENDING',
    total_amount DECIMAL(15,2),
    discount_amount DECIMAL(15,2) DEFAULT 0,
    final_amount DECIMAL(15,2),
    contact_name VARCHAR(255) NULL,
    contact_phone VARCHAR(50) NULL,
    guest_note TEXT NULL,
    manager_status_note VARCHAR(1000) NULL,
    cancellation_policy_snapshot_json TEXT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (venue_id) REFERENCES venues(id),
    FOREIGN KEY (series_id) REFERENCES booking_series(id) ON DELETE SET NULL,
    FOREIGN KEY (coupon_id) REFERENCES venue_coupons(id) ON DELETE SET NULL,
    INDEX series_id (series_id)
);

CREATE TABLE user_notifications (
    id CHAR(36) PRIMARY KEY,
    user_id CHAR(36) NOT NULL,
    type VARCHAR(50) NOT NULL DEFAULT 'BOOKING',
    title VARCHAR(255) NOT NULL,
    body TEXT NULL,
    metadata_json TEXT NULL,
    is_read TINYINT(1) NOT NULL DEFAULT 0,
    is_deleted TINYINT(1) NOT NULL DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_notifications_user_created (user_id, created_at),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE booking_items (
    id CHAR(36) PRIMARY KEY,
    booking_id CHAR(36),
    court_id CHAR(36),
    start_time DATETIME,
    end_time DATETIME,
    final_price DECIMAL(15,2),
    status VARCHAR(50),
    FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE CASCADE,
    FOREIGN KEY (court_id) REFERENCES courts(id)
);

CREATE TABLE court_blocks (
    id CHAR(36) PRIMARY KEY,
    court_id CHAR(36),
    start_time DATETIME,
    end_time DATETIME,
    created_by CHAR(36),
    FOREIGN KEY (court_id) REFERENCES courts(id) ON DELETE CASCADE,
    FOREIGN KEY (created_by) REFERENCES users(id)
);

-- PAYMENTS

CREATE TABLE payments (
    id CHAR(36) PRIMARY KEY,
    booking_id CHAR(36),
    method VARCHAR(50),
    status VARCHAR(50) DEFAULT 'PENDING',
    amount DECIMAL(15,2),
    gateway_reference VARCHAR(255),
    confirmed_by CHAR(36),
    confirmed_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (booking_id) REFERENCES bookings(id),
    FOREIGN KEY (confirmed_by) REFERENCES users(id)
);

CREATE TABLE refund_requests (
    id CHAR(36) PRIMARY KEY,
    booking_id CHAR(36),
    user_id CHAR(36),
    reason_code VARCHAR(50) NOT NULL DEFAULT 'PLAYER_CANCEL' COMMENT 'PLAYER_CANCEL|MANAGER_CANCEL|ADMIN',
    status VARCHAR(50) DEFAULT 'PENDING' COMMENT 'PENDING_RECONCILIATION|PENDING_REFUND|COMPLETED|REJECTED',
    requested_amount DECIMAL(15,2) NULL COMMENT 'Số tiền hoàn dự kiến theo snapshot',
    paid_amount DECIMAL(15,2) NULL COMMENT 'Số tiền đã thu (snapshot lúc tạo yêu cầu)',
    refund_bank_name VARCHAR(100) NULL COMMENT 'NH nhận hoàn (Player nhập)',
    refund_account_number VARCHAR(50) NULL,
    refund_account_holder VARCHAR(255) NULL,
    player_note TEXT NULL,
    rejection_reason TEXT NULL,
    manager_note TEXT NULL,
    manager_evidence_file_id CHAR(36) NULL COMMENT 'Ảnh bill CK hoàn từ Manager',
    player_received_file_id CHAR(36) NULL COMMENT 'Ảnh xác nhận nhận tiền từ Player (tùy chọn)',
    requested_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    processed_by CHAR(36),
    processed_at DATETIME,
    FOREIGN KEY (booking_id) REFERENCES bookings(id),
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (processed_by) REFERENCES users(id),
    FOREIGN KEY (manager_evidence_file_id) REFERENCES files(id) ON DELETE SET NULL,
    FOREIGN KEY (player_received_file_id) REFERENCES files(id) ON DELETE SET NULL,
    INDEX idx_refund_requests_status (status),
    INDEX idx_refund_requests_booking (booking_id)
);

CREATE TABLE refund_transactions (
    id CHAR(36) PRIMARY KEY,
    refund_request_id CHAR(36),
    payment_id CHAR(36),
    amount DECIMAL(15,2),
    method VARCHAR(50),
    status VARCHAR(50),
    gateway_reference VARCHAR(255),
    processed_by CHAR(36),
    processed_at DATETIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (refund_request_id) REFERENCES refund_requests(id),
    FOREIGN KEY (payment_id) REFERENCES payments(id),
    FOREIGN KEY (processed_by) REFERENCES users(id)
);

CREATE TABLE matching_posts (
    id CHAR(36) PRIMARY KEY,
    creator_user_id CHAR(36),
    booking_id CHAR(36),
    title VARCHAR(255),
    play_date DATE,
    play_start_time TIME,
    play_end_time TIME,
    venue_id CHAR(36),
    court_name VARCHAR(100),
    price_per_slot DECIMAL(15,2),
    required_players INT,
    skill_level VARCHAR(50),
    gender_pref VARCHAR(50),
    expense_sharing VARCHAR(100),
    play_purpose VARCHAR(100),
    notes TEXT,
    status VARCHAR(50) DEFAULT 'OPEN',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (creator_user_id) REFERENCES users(id),
    FOREIGN KEY (booking_id) REFERENCES bookings(id),
    FOREIGN KEY (venue_id) REFERENCES venues(id),
    INDEX idx_matching_posts_status_date (status, play_date),
    INDEX idx_matching_posts_creator (creator_user_id)
);

CREATE TABLE matching_post_items (
    id CHAR(36) PRIMARY KEY,
    post_id CHAR(36) NOT NULL,
    booking_item_id CHAR(36) NOT NULL,
    FOREIGN KEY (post_id) REFERENCES matching_posts(id) ON DELETE CASCADE,
    FOREIGN KEY (booking_item_id) REFERENCES booking_items(id) ON DELETE CASCADE,
    UNIQUE KEY uq_post_booking_item (post_id, booking_item_id)
);

CREATE TABLE matching_join_requests (
    id CHAR(36) PRIMARY KEY,
    post_id CHAR(36),
    user_id CHAR(36),
    message TEXT NULL,
    status VARCHAR(50) DEFAULT 'PENDING',
    reject_reason TEXT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES matching_posts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id),
    INDEX idx_matching_join_post_status (post_id, status)
);

CREATE TABLE matching_members (
    id CHAR(36) PRIMARY KEY,
    post_id CHAR(36),
    user_id CHAR(36),
    joined_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (post_id) REFERENCES matching_posts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id),
    UNIQUE KEY uq_matching_member (post_id, user_id)
);

CREATE TABLE matching_post_comments (
    id CHAR(36) PRIMARY KEY,
    post_id CHAR(36) NOT NULL,
    parent_comment_id CHAR(36) NULL,
    user_id CHAR(36) NOT NULL,
    content TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NULL,
    is_deleted TINYINT(1) NOT NULL DEFAULT 0,
    deleted_at DATETIME NULL,
    deleted_by_user_id CHAR(36) NULL,
    attachment_file_id CHAR(36) NULL,
    FOREIGN KEY (post_id) REFERENCES matching_posts(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(id),
    CONSTRAINT matching_post_comments_parent_fk FOREIGN KEY (parent_comment_id) REFERENCES matching_post_comments(id) ON DELETE SET NULL,
    CONSTRAINT matching_post_comments_deleted_by_fk FOREIGN KEY (deleted_by_user_id) REFERENCES users(id) ON DELETE SET NULL,
    CONSTRAINT matching_post_comments_attachment_fk FOREIGN KEY (attachment_file_id) REFERENCES files(id) ON DELETE SET NULL,
    INDEX idx_matching_comments_post (post_id, created_at),
    INDEX idx_matching_comments_post_active (post_id, is_deleted, created_at)
);

CREATE TABLE venue_reviews (
    id CHAR(36) PRIMARY KEY,
    venue_id CHAR(36),
